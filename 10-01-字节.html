    # 三数之和

    // 找出三数之和为0的情况
var threeSum = function(nums) {
    let ans = []
    const len = nums.length
    if(nums==null || len<3)return []
    nums.sort((a,b)=> a-b)// 排序降低时间复杂度
    for(let i=0;i<len;i++){
        if(nums[i]>0)break // 若当前数大于0则返回
        // 去重
        if(i>0 && nums[i] == nums[i-1])continue
        let L = i+1
        let R = len-1
        while(L<R){
            const sum = nums[i]+nums[L]+nums[R]
            if(sum == 0){
                ans.push([nums[i],nums[L],nums[R]])
                // 去重
                while(L<R && nums[L] == nums[L+1])L++
                while(L<R && nums[R] == nums[R-1])R--
                L++
                R--
            }
            else if(sum<0)L++
            else if(sum>0)R--
        }
    }
    return ans
};

    function listToTree(data) {
  let temp = {};// 存储对象
  let treeData = [];
  for (let i = 0; i < data.length; i++) {
    temp[data[i].id] = data[i];//{id:对象}的形式
  }
  for (let i in temp) {
    if (+temp[i].parentId != 0) {
      if (!temp[temp[i].parentId].children) {
        temp[temp[i].parentId].children = [];
      }// 给含有子节点的结点初始化children
      // push到父结点中
      temp[temp[i].parentId].children.push(temp[i]);
    } else {
      treeData.push(temp[i]);//根结点对象push到总数组中
    }
  }
  return treeData;
}
    1. 树形转列表：dfs+优先队列
    function treeToList(data) {
  let res = [];
  const dfs = (tree) => {
    tree.forEach((item) => {
      if (item.children) {
        dfs(item.children);
        delete item.children;
      }
      res.push(item);
    });
  };
  dfs(data);
  return res;
}
    1. 虚拟 Dom 转换成真实 Dom
    // 真正的渲染函数
function _render(vnode) {
  // 如果是数字类型转化为字符串
  if (typeof vnode === "number") {
    vnode = String(vnode);
  }
  // 字符串类型直接就是文本节点
  if (typeof vnode === "string") {
    return document.createTextNode(vnode);
  }
  // 普通DOM
  const dom = document.createElement(vnode.tag);
  if (vnode.attrs) {
    // 遍历属性
    Object.keys(vnode.attrs).forEach((key) => {
      const value = vnode.attrs[key];
      dom.setAttribute(key, value);
    });
  }
  // 子数组进行递归操作
  vnode.children.forEach((child) => dom.appendChild(_render(child)));
  return dom;
}

    // compose实现，合并多个函数的功能
function compose(...func){
  if(!func.length)return (v)=>v
  if(func.length==1)return fn[0]
  return fn.reduce(
    (pre,cur)=>
      (...args)=>
        pre(cur(...args))
  )
}

    给定一棵二叉树，二叉树每个节点只有0/2个孩子

    需要对每个节点赋一个值，使得左右子树权值和相等

    需要返回所有节点的最小权值和对10**9+7取模的结果

    def solution(root:TreeNode):
  res = [root]
  ans = 0
  while len(res)!=0:
    for i in range(len(res)):
      node = res.pop(0)
      if node.left:
        res.append(node.left)
      if node.right:
        res.append(node.right)
    ans += 1
    ans = (2**ans-1)%(10**9+7)
    return res

    def solution(string)
	dict1=[0 for _ in range(26)]
	a=string
	for i in a:
		dict1[ord(i)-ord('a')]+=1
	zero_num=0

			zero_num+=1
